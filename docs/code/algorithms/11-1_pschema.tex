\begin{algorithm}[H]
    \DontPrintSemicolon
    \SetAlgoVlined
    \SetKwInput{KwIn}{Input parameters}
    \KwIn{\Block{
            \g:\ Graph[\VertSet, \EdgeSet] \;
            \lbl:\ \LabelSet \;
        }}
    \KwOut{\Block{\vertex: \VertSet}}
    \blockskip

    \texttt{initialMessage} = \texttt{None} \;
    \texttt{v} = \texttt{Pregel(initialMessage,sendMsg,aggregateMsg,vProg)} \;
    \Return{v} \;
    \blockskip

    \DefInline{sendMsg(\lbl:\ \LabelSet, \g:\ Graph[\VertSet, \EdgeSet])}{
        \msgs\ where\ \ForEach{\lbl\ $\in$ \LabelSet}{
            \msgs.insert(
            \begin{math}
                \begin{cases}
                    \text{validate(\lbl, \g)} & \text{if}\ \lbl = \TripleConstraint \\
                    \text{validate(\lbl, \g)} & \text{if}\ \lbl = \ShapeReference   \\
                    \texttt{None}             & \text{otherwise}                    \\
                \end{cases}
            \end{math}
            )
        }
    }
    \blockskip

    \DefInline{aggregateMsg(\msgs:\ \MsgSet)}{
        \msgs\ where \;
        \ \ \ \msg\ =\
        \begin{math}
            \begin{cases}
                \msg      & \text{if}\ \msg \neq \texttt{None} \\
                \emptyset & \text{otherwise}                   \\
            \end{cases}
        \end{math}
    }
    \blockskip

    \DefInline{vProg(\lbl:\ \LabelSet, \g:\ Graph[\VertSet, \EdgeSet], \msgs:\ \MsgSet)}{
        \msgs\ where \;
        \ \ \ \msg\ =\
        \begin{math}
            \begin{cases}
                \text{validate(\lbl, \msgs, \g)} & \text{if}\ \lbl = \ShapeAnd    \\
                \text{validate(\lbl, \msgs, \g)} & \text{if}\ \lbl = \ShapeOr     \\
                \text{validate(\lbl, \msgs, \g)} & \text{if}\ \lbl = \Cardinality \\
                \texttt{None}                    & \text{otherwise}               \\
            \end{cases}
        \end{math}
    }\\
\end{algorithm}
